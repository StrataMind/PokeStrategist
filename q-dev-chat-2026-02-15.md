```typescript
// Algorithm
function calculateTypeCoverage(team: Pokemon[]) {
  const coverage = {
    offensive: {}, // How many types team can hit super-effectively
    defensive: {}   // How many weaknesses team has
  };
  
  team.forEach(pokemon =&gt; {
    pokemon.types.forEach(type =&gt; {
      // Calculate what this type can hit
      getSuperEffective(type).forEach(targetType =&gt; {
        coverage.offensive[targetType] = 
          (coverage.offensive[targetType] || 0) + 1;
      });
      
      // Calculate what this type is weak to
      getWeaknesses(type).forEach(attackType =&gt; {
        coverage.defensive[attackType] = 
          (coverage.defensive[attackType] || 0) + 1;
      });
    });
  });
  
  return coverage;
}
```

### **Synergy Score Algorithm**

```typescript
function calculateSynergy(team: Pokemon[]) {
  let score = 0;
  
  // Type diversity (max 20 points)
  const uniqueTypes = new Set(team.flatMap(p =&gt; p.types));
  score += Math.min(uniqueTypes.size * 2, 20);
  
  // Stat balance (max 20 points)
  const avgPhysical = average(team.map(p =&gt; p.stats.attack));
  const avgSpecial = average(team.map(p =&gt; p.stats.specialAttack));
  const balance = 1 - Math.abs(avgPhysical - avgSpecial) / 200;
  score += balance * 20;
  
  // Coverage (max 30 points)
  const coverage = calculateTypeCoverage(team);
  const coverageScore = Object.keys(coverage.offensive).length;
  score += Math.min(coverageScore * 1.67, 30);
  
  // Weakness coverage (max 30 points)
  const maxWeaknesses = Math.max(...Object.values(coverage.defensive));
  const weaknessScore = (6 - maxWeaknesses) * 5;
  score += Math.max(weaknessScore, 0);
  
  return Math.round(score); // 0-100
}
```

### **Damage Calculator**

```typescript
function calculateDamage(
  attacker: Pokemon,
  defender: Pokemon,
  move: Move,
  options: {
    weather?: Weather;
    terrain?: Terrain;
    criticalHit?: boolean;
  }
) {
  // Base damage formula
  const level = 50; // VGC standard
  const attack = move.category === &#39;Physical&#39; 
    ? attacker.stats.attack 
    : attacker.stats.specialAttack;
  const defense = move.category === &#39;Physical&#39;
    ? defender.stats.defense
    : defender.stats.specialDefense;
  
  let damage = ((2 * level / 5 + 2) * move.power * attack / defense / 50 + 2);
  
  // Modifiers
  const stab = attacker.types.includes(move.type) ? 1.5 : 1;
  const effectiveness = getTypeEffectiveness(move.type, defender.types);
  const random = 0.85; // Minimum roll
  const critical = options.criticalHit ? 1.5 : 1;
  
  damage *= stab * effectiveness * random * critical;
  
  return {
    min: Math.floor(damage * 0.85),
    max: Math.floor(damage),
    percentage: (damage / defender.stats.hp) * 100
  };
}
```
